fun main(args: Array<String>) {
    runApplication<ColorServiceApplication>(*args)

@Document
data class Color(
        @Id val id: String,
        val hexCode: String,
        val name: String
)
interface ColorRepo : ReactiveMongoRepository<Color, String>
                

@SpringBootApplication
class ColorServiceApplication(val objectMapper: ObjectMapper, val colorRepo: ColorRepo) {

    @EventListener(ApplicationReadyEvent::class)
    fun injectColors() {
        val classLoader = javaClass.classLoader
        val path = classLoader.getResource("color.json").file
        Mono.from(colorRepo.deleteAll())
                .thenReturn(objectMapper.readValue<List<Color>>(File(path)))
                .flatMapMany { colorRepo.insert(it) }
                .subscribe { logger.info(it) }
    }

    companion object {
        private val logger = LogManager.getLogger(this.javaClass.enclosingClass)
    }
}

Backup: REST API


@Configuration
class ColorRouter(val colorHandler: ColorHandler) {
    @Bean
    fun route() = router {
        "/colors".nest {
            GET("/", colorHandler::findAllForHttp)
        }

    }
}
                

@Component
class ColorHandler(val colorRepo: ColorRepo) {
    fun findAllForHttp(req: ServerRequest): Mono<ServerResponse> {
        return findAll()
                .`as` { ok().json().body(it) }
    }

    fun findAll() = colorRepo.findAll()
            .delayElements(Duration.ofMillis(50))
}

Backup: Rsocket


@Configuration
@Component
class ColorRSocketController(val colorHandler: ColorHandler, val objectMapper: ObjectMapper) : ApplicationListener<ApplicationReadyEvent>,
        SocketAcceptor {
                ...
    override fun onApplicationEvent(event: ApplicationReadyEvent) {
        logger.info("Opening the rsocket on port 8081")
        RSocketFactory
                .receive()
                .acceptor(this)
                .transport(TcpServerTransport.create(8081))
                .start()
                .subscribe()
    }

    companion object {
        private val logger = LogManager.getLogger(this.javaClass.enclosingClass)

    }
}
                

    override fun accept(setup: ConnectionSetupPayload?, sendingSocket: RSocket?): Mono<RSocket> {
        val abstractRSocket = object : AbstractRSocket() {
            override fun requestStream(payload: Payload?): Flux<Payload> {
                return colorHandler.findAll()
                        .map { objectMapper.writeValueAsString(it) }
                        .map { DefaultPayload.create(it) }
            }
        }
        return Mono.just(abstractRSocket)
    }
